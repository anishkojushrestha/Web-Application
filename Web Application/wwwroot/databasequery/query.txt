CREATE TABLE CompanyInfo(
CompanyId bigint NOT NULL primary key,
CompanyName varchar(50),
Address varchar(50),
Email varchar(50),
PanNumber int,
City varchar(50),
Coutry varchar(50),
AchiveFrom date,
AchiveTo date,

)

CREATE TABLE ContactPerson(
ContactId bigint NOT NULL primary key,
ContactName varchar(50),       
PhoneNumber int,
MobileNumber int,
Designation varchar(50)
CompanyId int foreign key References companyinfo(Id)

)

CREATE TABLE User(
UserId bigint NOT NULL primary key,,
FirstName varchar(50),
LastName varchar(50),
UserName varchar(50),
Address varchar(50),
Password varchar(50),
Profile varchar(50),
IsActive bool,
)
